SDK for Every Language

Build AI-powered applications with native SDKs for Python, Go, and TypeScript

prompt.py ‚Äî Start a conversation 
from langdag import LangDAGClient

with LangDAGClient() as client:
    # Start a conversation
    node = client.prompt(
        message="What is LangDAG?"
    )
    print(node.content)

    # Continue from any node
    node2 = node.prompt("Tell me more")
